#include<bits/stdc++.h>
using namespace std;
using ll=long long;
const int inf=1e10;
 
void solve(){
	int n;
	cin>>n;
	string s;
	cin>>s;
	for(int k=n;k>=1;k--){
		vector<int> v(k);
		for(int i=0;i<n;++i){
			v[i%k]^='1'-s[i];
		}
		if(v==vector(k,v[0])){
			cout<<k<<'\n';
			return;
		}
	}
}
 
int main(){
	int T;
	cin>>T;
	while(T--){
		solve();
	}
}
